def create_year2_journey_map():

def create_knowledge_map():
    """Create a visualization of acquired knowledge."""

    categories = {
        'QEC Theory': {
            'topics': ['Error models', 'Stabilizer codes', 'CSS construction',
                      'Surface codes', 'Color codes', 'QLDPC'],
            'level': [5, 5, 5, 5, 4, 4]
        },

            'topics': ['FT definitions', 'Magic states', 'Distillation',
                      'Transversal gates', 'Code switching', 'Resources'],
            'level': [5, 5, 4, 5, 4, 4]
        },
        'Hardware': {
            'topics': ['Superconducting', 'Trapped ions', 'Neutral atoms',
                      'Photonics', 'Control', 'Readout'],
            'level': [4, 4, 4, 3, 4, 4]
        },
        'Algorithms': {
            'topics': ['VQE', 'QAOA', 'Phase estimation', 'Error mitigation',
                      'Complexity', 'QML'],
            'level': [5, 5, 4, 4, 4, 3]
        }
    }

    fig, ax = plt.subplots(figsize=(14, 10))

    colors = {'QEC Theory': 'steelblue', 'Fault Tolerance': 'coral',
             'Hardware': 'seagreen', 'Algorithms': 'mediumpurple'}

    y_offset = 0
    for cat_name, cat_data in categories.items():
        n_topics = len(cat_data['topics'])
        for i, (topic, level) in enumerate(zip(cat_data['topics'], cat_data['level'])):
            ax.barh(y_offset, level, color=colors[cat_name], alpha=0.7,
                   edgecolor='black', height=0.8)
            ax.text(-0.3, y_offset, topic, ha='right', va='center', fontsize=9)
            y_offset += 1
        ax.axhline(y_offset - 0.5, color='gray', linestyle='-', alpha=0.3)
        ax.text(5.5, y_offset - n_topics/2 - 0.5, cat_name, ha='left', va='center',
               fontsize=12, fontweight='bold', color=colors[cat_name])

    ax.set_xlim(-4, 7)
    ax.set_ylim(-1, y_offset)
    ax.set_xlabel('Mastery Level (1-5)', fontsize=12)
    ax.set_title('Year 2 Knowledge Acquisition Map', fontsize=14)
    ax.set_xticks([1, 2, 3, 4, 5])
    ax.set_yticks([])

    # Add level markers
    for level in [1, 2, 3, 4, 5]:
        ax.axvline(level, color='gray', linestyle=':', alpha=0.3)

    plt.tight_layout()
    plt.savefig('year2_knowledge_map.png', dpi=150, bbox_inches='tight')
    plt.show()

    return fig
